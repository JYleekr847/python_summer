void printboard(int n);

int place(int k,int i);
int x[100];
int count;

void NQueen(int k,int n)
{
  int i;
  for(i=1;i<=n;i++)
  {
    if(place(k,i)==1)
    {     x[k]=i;
            if(k==n)
            {   
                count++;
                printf("Solution %d: \n ", count);
                printboard(n);
            }
            else
                NQueen(k+1,n);
    }
  }
}

int place(int k,int i)
{
  int j;
  for(j=1;j<k;j++)
  {
    if((x[j]==i)||abs(x[j]-i)==abs(j-k))
        return 0;
  }
  return 1;
}

void printboard(int n)
{
  int i;
  for(i=1;i<=n;i++)
    printf("%d  ",x[i]);
  printf("\n");
}

int main()
{
    int n;
    
    clock_t begin, end;
    begin = clock();
    
    printf("Enter Value of N:");
    scanf("%d",&n);
    NQueen(1,n);
    
    end = clock();
    float excution_time = (float)(end - begin)/CLOCKS_PER_SEC;
    
    printf(" code excution time : %f sec", excution_time);
}
